```dart
void main() {
  group('TotpService', () {
    // RFC 6238 test vector: SHA1, secret "12345678901234567890", 30s period
    test('generates correct 6-digit TOTP for RFC test vector', () {
      final service = TotpService();
      final code = service.generate(
        base32Secret: 'GEZDGNBVGY3TQOJQGEZDGNBVGY3TQOJQ', // "12345678901234567890"
        timestampMs: 59000, // exactly 59s, time step 1
      );
      expect(code, hasLength(6));
    });

    test('same timestamp produces same code', () {
      final service = TotpService();
      final a = service.generate(base32Secret: 'GEZDGNBVGY3TQOJQGEZDGNBVGY3TQOJQ', timestampMs: 59000);
      final b = service.generate(base32Secret: 'GEZDGNBVGY3TQOJQGEZDGNBVGY3TQOJQ', timestampMs: 59000);
      expect(a, equals(b));
    });

    test('different timestamps produce different codes', () {
      final service = TotpService();
      final a = service.generate(base32Secret: 'GEZDGNBVGY3TQOJQGEZDGNBVGY3TQOJQ', timestampMs: 59000);
      final b = service.generate(base32Secret: 'GEZDGNBVGY3TQOJQGEZDGNBVGY3TQOJQ', timestampMs: 59000 + 30000);
      expect(a, isNot(equals(b)));

    test('parseOtpauthUrl extracts secret from url', () {
      final result = TotpService.parseOtpauthUrl(

      expect(result.label, 'Example:alice@google.com');
      expect(result.issuer, 'Example');
    });

    test('formatCode inserts space for readability', () {
      expect(TotpService.formatCode('482931'), '482 931');
      expect(TotpService.formatCode('48720691'), '4872 0691');
    });
  });
```

```dart
// lib/core/password_generator/totp_service.dart
import 'package:otp/otp.dart';

class ParsedOtpAuth {
  const ParsedOtpAuth({required this.secret, this.label, this.issuer});
  final String secret;
  final String? label;
  final String? issuer;
}

class TotpService {
  String generate({
    required String base32Secret,
    required int timestampMs,
    int length = 6,
    int interval = 30,
    Algorithm algorithm = Algorithm.SHA1,
  }) {
    return OTP.generateTOTPCodeString(
      base32Secret,
      timestampMs,
      length: length,
      interval: interval,
      algorithm: algorithm,
    );
  }

  int remainingSeconds(int timestampMs, {int interval = 30}) {
    return interval - (timestampMs ~/ 1000 % interval);
  }

  static ParsedOtpAuth parseOtpauthUrl(String url) {
    final uri = Uri.parse(url);
    final secret = uri.queryParameters['secret'] ?? '';
    String? label;
    if (uri.pathSegments.isNotEmpty) {
      label = Uri.decodeComponent(uri.pathSegments.first);
    }
    final issuer = uri.queryParameters['issuer'];
    return ParsedOtpAuth(secret: secret, label: label, issuer: issuer);
  }

  static String formatCode(String code) {
    if (code.length <= 4) return code;
    final mid = code.length ~/ 2;
    return '${code.substring(0, mid)} ${code.substring(mid)}';
  }
}
```

```dart
// lib/features/totp/totp_page.dart
import 'dart:async';
import 'package:flutter/material.dart';
import 'package:secure_box/app/app_services.dart';
import 'package:secure_box/core/password_generator/totp_service.dart';

class TotpPage extends StatefulWidget {
  const TotpPage({super.key, required this.services});
  final AppServices services;

  @override
  State<TotpPage> createState() => _TotpPageState();

class _TotpPageState extends State<TotpPage> {
  final TotpService _totpService = TotpService();
  List<VaultListItem> _items = [];
  Timer? _timer;
  int _tick = 0;

  @override
  void initState() {
    super.initState();
    _loadItems();
    _timer = Timer.periodic(const Duration(seconds: 1), (_) {
      if (mounted) setState(() => _tick++);

  @override
  void dispose() {
    _timer?.cancel();
    super.dispose();
  }

  Future<void> _loadItems() async {
    final items = await widget.services.listVaultItems();
    if (!mounted) return;
    setState(() => _items = items);
  }

  // Filter items with TOTP secret. We need to get full entries for this.
  // In practice, use listVaultItems and filter by those with totpSecret.
  // For now, show all items as placeholder. Full integration comes in Task 4.

  @override
  Widget build(BuildContext context) {
    final now = DateTime.now().millisecondsSinceEpoch;
    return SecureVisualBackground(
      child: RefreshIndicator(
        onRefresh: _loadItems,
        child: _items.isEmpty
            ? _buildEmpty(context)
            : _buildGrid(context, now),
      ),
    );
  }

  Widget _buildEmpty(BuildContext context) {
    return Center(
      child: Column(
        mainAxisSize: MainAxisSize.min,
        children: [
          Icon(Icons.qr_code_2_rounded, size: 64, color: Theme.of(context).colorScheme.primary),
          const SizedBox(height: 12),
          Text('暂无疑问验证码条目', style: Theme.of(context).textTheme.titleMedium),
          const SizedBox(height: 4),
          Text('编辑条目时添加 TOTP 密钥即可在此查看', style: Theme.of(context).textTheme.bodyMedium),
        ],
      ),
    );
  }

  Widget _buildGrid(BuildContext context, int nowMs) {
    return GridView.builder(
      padding: const EdgeInsets.all(16),
      gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
        crossAxisCount: 2,
        mainAxisSpacing: 12,
        crossAxisSpacing: 12,
        childAspectRatio: 0.85,
      ),
      itemCount: _items.length,
      itemBuilder: (context, index) => _TotpCard(
        item: _items[index],
        nowMs: nowMs,
        totpService: _totpService,
      ),
    );
  }
}

class _TotpCard extends StatelessWidget {
  const _TotpCard({required this.item, required this.nowMs, required this.totpService});
  final VaultListItem item;
  final int nowMs;
  final TotpService totpService;

  @override
  Widget build(BuildContext context) {
    final remaining = totpService.remainingSeconds(nowMs);
    final progress = remaining / 30.0;
    final color = remaining > 10
        ? SecureVisualColors.success
        : remaining > 5
            ? const Color(0xFFF5A623)
            : SecureVisualColors.danger;

    return GestureDetector(
      onTap: () {
        // Copy TOTP code to clipboard
        final code = totpService.generate(
          base32Secret: 'placeholder', // Will use actual secret from entry
          timestampMs: nowMs,
        );
        // Clipboard.setData + ScaffoldMessenger snackbar
      },
      child: SecureGlassCard(
        padding: const EdgeInsets.all(16),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            SizedBox(
              width: 56, height: 56,
              child: CircularProgressIndicator(value: progress, color: color, strokeWidth: 3),
            ),
            const SizedBox(height: 12),
            Text('--- ---', style: const TextStyle(fontSize: 24, fontWeight: FontWeight.w800, letterSpacing: 4)),
            const SizedBox(height: 8),
            Text(item.title, style: Theme.of(context).textTheme.bodySmall?.copyWith(fontWeight: FontWeight.w600)),
            Text(item.username, style: Theme.of(context).textTheme.bodySmall, overflow: TextOverflow.ellipsis),
            const SizedBox(height: 4),
            Text('${remaining}s', style: TextStyle(fontSize: 11, color: color)),
          ],
        ),
      ),
    );
  }
}
```

```dart
class TotpListItem {
  TotpListItem({required this.id, required this.title, required this.username, required this.totpSecret});
  final String id;
  final String title;
  final String username;
  final String totpSecret;
}

Future<List<TotpListItem>> listTotpItems() async {
  _ensureUnlocked();
  final items = await repository.itemsDao.allItemsForManifest();
  final active = items.where((i) => i.deletedAt == null).toList();
  return _session.withDekCopy((dek) {
    return active
        .map((item) {
          final entry = _decryptItemAsEntry(item, dek);
          if (entry.totpSecret == null || entry.totpSecret!.isEmpty) return null;
          return TotpListItem(
            id: item.id,
            title: entry.title,
            username: entry.username,
            totpSecret: entry.totpSecret!,
          );
        })
        .whereType<TotpListItem>()
        .toList();
  });
}
```
